实际活跃的物联网设备各省份分布数 资产业务 国内活跃物联网设备数-疑似蜜罐设备数(维度:省份,月份) assets_detail select
count(distinct vendor)
from hive_xlkh.assets_inofo where id_batch like ‘202312%’ and dst_country = ‘中国’; - 0
探测组织数 嗅探业务 对探测组织进行计数(维度:月份，探测组织) group_scan_detail select
count(distinct group_name) as 探测组织数 from hive_xlkh.group_scan_detail where id_batch like ‘202312%’;
探测组织节点数 嗅探业务 对探测组织IP个数进行计数(维度:月份，探测组织) group_scan_detail select
group_name as 探测组织, count(distinct src_ip) as 探测组织节点数
from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ group by group_name;
扫描探测境内IP数 嗅探业务 对所有的探测IP进行去重统计(维度:月份) group_scan_detail select
count(distinct dst_ip) as 探测境内IP数
from hive_xlkh.group_scan_detail where id_batch like 202312% and dst_country = ‘中国’;
探测事件数 嗅探业务 对所有的探测事件进行不去重统计(维度:月份) group_scan_detail select
count(1) as 探测事件数
from hive_xlkh.group_scan_detail where id_batch like 202312%;
境外组织探测端口数 嗅探业务 对所有探测目标端口进行统计(维度:月份) group_scan_detail select
count(distinct dst_port) as 探测端口数
from hive_xlkh.group_scan_detail where id_batch like 202312%
被扫描IP地理位置分布情况 嗅探业务 对被探测的IP进行统计(维度:月份，省份) group_scan_detail select
src_country,
count(1) as num from hive_xlkh.group_scan_detail where id_batch like 202312% and dst_country = 中国 group by src_country order by desc num;
探测组织节点地理位置分布(图-3) 嗅探业务 对探测组织的IP分布进行统计(维度:国家) group_scan_detail select
src_country as 探测国家, count(1) as 探测次数
from hive_xlkh.group_scan_detail where id_batch like 202312% group by src_country desc num;
重点组织的探测活跃情况(表-3) 嗅探业务 对重点组织的活跃情况做统计(指标: 探测事件个数,节点数量,目的IP个数,探测端口数量,目的省份个数) group_scan_detail select
group_name as 组织名称, count(1) as 探测事件个数, count(distinct src_ip) as 节点数量,
count(distinct dst_ip) as 目的IP数量,
count(distinct dst_port) as 探测端口数量, count(distinct dst_province) as 目标省份个数 from hive_xlkh.group_scan_detail where id_batch like 202312% and dst_country = 中国
group by group_name;
活跃探测节点top10(表-4) 嗅探业务 对探测组织的探测数量进行求topN(维度:月份，国家,探测组织) group_scan_detail select
src_ip as 探测节点, group_name as 所属组织, src_country as 节点位置, count(1) as 探测事件数
from hive_xlkh.group_scan_detail where id_batch like 202312% group by src_iph, group_name, src_country
order by 探测事件数 desc
limit 10;
top10探测节点探测事件占比 嗅探业务 top10探测节点探测事件总数/探测事件总数 group_scan_detail sum(top10)/select count(1) from from hive_xlkh.group_scan_detail where id_batch like 202312%;
探测组织节点数 嗅探业务 针对特定探测组织求节点数(维度: 探测组织) group_scan_detail select
count(distinct dst_ip) as 探测组织节点数
from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ and group_name = ‘Shadowserver’;
扫描探测端口数 嗅探业务 统计国内被扫描端口数量(维度:国家) group_scan_detail select
count(distinct dst_port) as 扫描探测端口数
from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ and group_name = ‘Shadowserver’;
扫描探测境内IP数 嗅探业务 判定探测事件ip为境内IP(维度: 国家) group_scan_detail select
count(distinct dst_ip) as 扫描探测境内IP数
from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ and group_name = ‘Shadowserver’ and dst_country = ‘中国’;
覆盖省份数 嗅探业务 被探测组织探测的境内IP的省份分布(维度:省份) group_scan_detail select
dst_province,
count(distinct dst_province) as 覆盖省份个数
from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ and group_name = ‘Shadowserver’ and dst_country = ‘中国’ group by dst_province;
探测节点活跃度Top10(表-5) 嗅探业务 group_scan_detail select
src_ip as 探测节点, src_country as 探测节点位置, count(1) as 探测事件, count(distinct dst_port) as 探测端口,
sum(case when dst_port BETWEEN 1 and 1023 then 1 ELSE 0 end) as 熟知端口数 from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ and group_name = ‘Shadowserver’ group by src_ip, src_country order by 探测事件 desc;

探测节点时间行为分析(图-4) 嗅探业务 group_scan_detail select
src_ip,
count(1) as num
from hive_xlkh.group_scan_detail where id_batch = ‘20231201’ group by src_ip
order by num desc;
探测节点协议行为分析(图-5) 嗅探业务 group_scan_detail select
src_ip,
protocol,
count(1) as num
from hive_xlkh.group_scan_detail where id_batch like ‘202312%’ group by src_ip, protocol
order by num desc;
监测行业资产数 行业业务 flow_security/Analysis.xsl
受攻击资产数 行业业务 flow_security select
count(distinct dst_ip) as 受攻击资产
from hive_xlkh.flow_security where id_batch like ‘202312%’ ;
受攻击资产数分布省份(图-6) 行业业务 flow_security select
dst_ip,
dst_province,
count(distinct dst_ip) as num from hive_xlkh.flow_security where id_batch like ‘202312%’ and dst_country = ‘中国’
group by dst_ip, dst_province order by num desc;
攻击事件数 行业业务 flow_security select
count(1) as 攻击事件数
from hive_admin.flow_security id_batch like ‘202312%’ and id_rule = ‘xxx’;
行业资产及攻击事件分布(表-6) 行业业务 flow_security 上面受监测资产,受攻击资产,攻击事件数 组合
攻击源国家个数 行业业务 flow_security select
src_country as 攻击源国家 from hive_admin.flow_security where id_batch like ‘202312%’ and dst_country = ‘中国’
group by src_country;
攻击源节点个数 行业业务 flow_security select
count(distinct src_ip) as 攻击源节点个数
from hive_admin.flow_security where id_batch like ‘202312%’ and dst_country = ‘中国’;
攻击源国家top5 行业业务 flow_security select
src_country as 攻击源国家, count(1) as 攻击事件数
from hive_admin.flow_security where id_batch like ‘202312%’ and dst_country = ‘中国’ group by src_country
order by 攻击事件数 desc
limit 5;
攻击类型占比(图-7) 行业业务 flow_security select
attack_type as 攻击类型, count(1) / (select count(1) as num from hive_admin.flow_security where id_batch like ‘202312%’ and dst_country = ‘中国’) as 攻击类型占比 from hive_admin.flow_security where id_batch like ‘202312%’ and dst_country = ‘中国’
特定攻击类型分析 行业业务 flow_security
通联资产数 行业业务 flow_detail select
count(1) as 通联资产数
from hive_xlkh.flow_detail where id_batch like ‘202312%’ and (src_country or dst_country = ‘中国’) and (src_country or dst_country = ‘中国’)
group by src_ip, dst_ip
通联频率 行业业务 flow_detail select
count(1) as 通联频数
from hive_xlkh.flow_detail where id_batch like ‘202312%’ and (src_country or dst_country = ‘中国’) and (src_country or dst_country = !’中国’)
通联国家top10 行业业务 flow_detail select
src_country,
dst_country,
sum(case when src_country or dst_country != ‘中国’ then 1 else 0 end) as 通联次数
from hive_xlkh.flow_detail where id_batch like ‘202312%’ and (src_country or dst_country = ‘中国’) and (src_country or dst_country = !’中国’)
攻击事件境外IPtop10(表-7) 行业业务 flow_security select
src_ip as 攻击ip,
count(1) as 攻击事件数量, attack_type as 攻击类型, count(1) / (select count(1) from hive_admin.flow_detail where id_batch like ‘202312%’ and dst_country != ‘中国’) as 攻击事件占比 from hive_admin.flow_security where id_batch like ‘202312%’ and src_country = ‘中国’ and dst_country != ‘中国’ group by src_ip order by 攻击事件数量 desc limit 10;
总攻击事件数 行业业务 flow_security select
count(1) as 总攻击事件数 from hive_admin.flow_security where id_batch like ‘202312%’ and dst_country != ‘中国’
访问事件境外IPtop10(表-8) 行业业务 flow_detail select
src_ip as 访问ip, dst_country as 境外国家, count(1) as 数据事件访问数, count(distinct dst_ip) as 访问资产数,
src_port as 端口
from hive_admin.flow_detail where id_batch like ‘202312%’ and dst_country != ‘中国’ and src_country = ‘中国’ group by src_ip, dst_country, src_port order by 数据事件访问数 desc limit 10;
